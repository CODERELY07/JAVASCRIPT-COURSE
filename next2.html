<script>
const msg = document.querySelector('.msg');
const inputName = document.getElementById('name');
const inputEmail= document.getElementById('email');
const inputSubmit = document.getElementById('submit');
const ul = document.querySelector('.list-items');
const users = document.querySelectorAll('.users');
inputSubmit.addEventListener('click', onSubmit);

function onSubmit(e) {
e.preventDefault();

const username = inputName.value.trim();
const email = inputEmail.value.trim();

// Check if the username and email are not empty
if (username === "" || email === "") {
    const errorMessage = document.createElement('span');
    errorMessage.classList.add('error');
    errorMessage.textContent = "Please enter all fields";
    msg.appendChild(errorMessage);

    setTimeout(() => errorMessage.remove(), 3000);
    return;
}

// Check if the username is unique
const existingUsernames = Array.from(ul.querySelectorAll('li')).map(li => li.textContent.split(' : ')[0].trim());
if (existingUsernames.includes(username)) {
    const errorMessage = document.createElement('span');
    errorMessage.classList.add('error');
    errorMessage.textContent = "Username already exists";
    msg.appendChild(errorMessage);

    setTimeout(() => errorMessage.remove(), 3000);
    return;
}

// Check if the email is unique
const existingEmails = Array.from(ul.querySelectorAll('li')).map(li => li.textContent.split(' : ')[1].trim());
if (existingEmails.includes(email)) {
    const errorMessage = document.createElement('span');
    errorMessage.classList.add('error');
    errorMessage.textContent = "Email already exists";
    msg.appendChild(errorMessage);

    setTimeout(() => errorMessage.remove(), 3000);
    return;
}

// If both username and email are unique, add the user to the list
const li = document.createElement('li');
li.textContent = `${username} : ${email}`;
ul.appendChild(li);

// Clear input fields
inputName.value = '';
inputEmail.value = '';
}

</script>
